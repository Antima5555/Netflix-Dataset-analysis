def categorize_col(x):
    if x<= df['Vote_Average'].quantile(0.25):
        return 'not_popular'
    elif x<= df['Vote_Average'].quantile(0.50):
        return 'below_average'
    elif x<= df['Vote_Average'].quantile(0.75):
        return 'average'
    else:
        return'popular'

df['Vote_Average']= df['Vote_Average'].apply(categorize_col)
